$ sudo yum install http://repo.zabbix.com/zabbix/2.2/rhel/7/x86_64/zabbix-release-2.2-1.el7.noarch.rpm
$ sudo yum install zabbix-server-mysql zabbix-web-mysql zabbix-web-japanese zabbix-agent

DB設定
$ mysql -u root -p
create database zabbix;
grant all privileges on zabbix.* to zabbix@localhost identified by 'password' ;
flush privileges;

$ sudo mysql -uroot -p zabbix < /usr/share/doc/zabbix-server-mysql-2.2.9/create/schema.sql
$ sudo mysql -uroot -p zabbix < /usr/share/doc/zabbix-server-mysql-2.2.9/create/images.sql
$ sudo mysql -uroot -p zabbix < /usr/share/doc/zabbix-server-mysql-2.2.9/create/data.sql
